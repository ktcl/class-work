# To change the title font size. remember ; gets rid of text in output Text(0.5,1,'Department Sales')
p = df.plot.line(figsize=(15,5))
p.set_title('Department Sales', fontsize=40);
p.set_xlabel('X Axis',fontsize=15);
p.set_ylabel('Y Axis',fontsize=15);
In [28]:

# To change the color of the plot line and the title. Colors are not case sensitive.
p = df.plot.line(figsize=(15,5), color='red')
p.set_title('Department Sales', fontsize=40, color='Purple');

p.set_ylabel('Y Axis',fontsize=15);
p.legend(['New Legend']);
